**Step 3: Write the autoloop.sh script**

Create `templates/.specdev/skills/tools/autoloop/scripts/autoloop.sh`:

```bash
#!/usr/bin/env bash
set -euo pipefail

# autoloop.sh — Invoke external CLI reviewer, parse pass/fail, enforce round limits
#
# Usage: autoloop.sh --reviewer <name> --round <N> [--scope diff|files|custom] [--context <text>]
# Env: AUTOLOOP_REVIEWERS_DIR — override reviewers directory (default: ../reviewers relative to this script)
# Output: JSON to stdout with verdict, round, max_rounds, escalate, findings
# Exit: 0 on successful execution (even if verdict=fail), 1 on error

REVIEWER=""
ROUND=""
SCOPE=""
CONTEXT=""

while [[ $# -gt 0 ]]; do
  case "$1" in
    --reviewer) REVIEWER="$2"; shift 2 ;;
    --round) ROUND="$2"; shift 2 ;;
    --scope) SCOPE="$2"; shift 2 ;;
    --context) CONTEXT="$2"; shift 2 ;;
    *) echo "Unknown argument: $1" >&2; exit 1 ;;
  esac
done

if [ -z "$REVIEWER" ]; then
  echo "Error: --reviewer is required" >&2
  echo "Usage: autoloop.sh --reviewer <name> --round <N>" >&2
  exit 1
fi

if [ -z "$ROUND" ]; then
  ROUND=1
fi

# Locate reviewers directory
SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
REVIEWERS_DIR="${AUTOLOOP_REVIEWERS_DIR:-${SCRIPT_DIR}/../reviewers}"

CONFIG_FILE="${REVIEWERS_DIR}/${REVIEWER}.json"
if [ ! -f "$CONFIG_FILE" ]; then
  echo "Error: reviewer config not found: ${CONFIG_FILE}" >&2
  exit 1
fi

# Parse JSON config (portable — no jq dependency)
CONFIG=$(cat "$CONFIG_FILE")

get_json_string() {
  echo "$CONFIG" | grep -o "\"$1\"[[:space:]]*:[[:space:]]*\"[^\"]*\"" | sed "s/\"$1\"[[:space:]]*:[[:space:]]*\"//" | sed 's/"$//'
}

get_json_number() {
  echo "$CONFIG" | grep -o "\"$1\"[[:space:]]*:[[:space:]]*[0-9]*" | sed "s/\"$1\"[[:space:]]*:[[:space:]]*//"
}

COMMAND=$(get_json_string "command")
CONFIG_SCOPE=$(get_json_string "scope")
MAX_ROUNDS=$(get_json_number "max_rounds")
PASS_PATTERN=$(get_json_string "pass_pattern")
FAIL_PATTERN=$(get_json_string "fail_pattern")

# Defaults
SCOPE="${SCOPE:-${CONFIG_SCOPE:-diff}}"
MAX_ROUNDS="${MAX_ROUNDS:-3}"
PASS_PATTERN="${PASS_PATTERN:-LGTM|no issues|approved|pass}"
FAIL_PATTERN="${FAIL_PATTERN:-needs changes|issues found|fail|reject}"

if [ -z "$COMMAND" ]; then
  echo "Error: reviewer config missing 'command' field" >&2
  exit 1
fi

# Assemble context based on scope
case "$SCOPE" in
  diff)
    REVIEW_CONTEXT=$(git diff 2>/dev/null || echo "")
    if [ -z "$REVIEW_CONTEXT" ]; then
      REVIEW_CONTEXT=$(git diff --cached 2>/dev/null || echo "")
    fi
    ;;
  files)
    CHANGED_FILES=$(git diff --name-only 2>/dev/null || echo "")
    if [ -z "$CHANGED_FILES" ]; then
      CHANGED_FILES=$(git diff --cached --name-only 2>/dev/null || echo "")
    fi
    REVIEW_CONTEXT=""
    while IFS= read -r f; do
      if [ -n "$f" ] && [ -f "$f" ]; then
        REVIEW_CONTEXT+="=== $f ===
"
        REVIEW_CONTEXT+="$(cat "$f")
"
      fi
    done <<< "$CHANGED_FILES"
    ;;
  custom)
    REVIEW_CONTEXT="$CONTEXT"
    ;;
  *)
    echo "Error: unknown scope: $SCOPE" >&2
    exit 1
    ;;
esac

# Build review prompt
REVIEW_PROMPT="Review the following code changes. If acceptable, reply with PASS. If issues found, describe them.

${REVIEW_CONTEXT}"

# Substitute placeholders in command
FINAL_CMD="${COMMAND//\{prompt\}/$REVIEW_PROMPT}"

# Collect changed file paths for {files} placeholder
if [ "$SCOPE" = "files" ] || [ "$SCOPE" = "diff" ]; then
  FILE_LIST=$(git diff --name-only 2>/dev/null || echo "")
  if [ -z "$FILE_LIST" ]; then
    FILE_LIST=$(git diff --cached --name-only 2>/dev/null || echo "")
  fi
else
  FILE_LIST=""
fi
FINAL_CMD="${FINAL_CMD//\{files\}/$FILE_LIST}"

# Execute reviewer command
REVIEWER_OUTPUT=""
if echo "$COMMAND" | grep -q '{stdin}'; then
  STDIN_CMD="${COMMAND//\{stdin\}/}"
  STDIN_CMD="${STDIN_CMD//\{prompt\}/$REVIEW_PROMPT}"
  REVIEWER_OUTPUT=$(echo "$REVIEW_CONTEXT" | eval "$STDIN_CMD" 2>&1) || {
    echo "Error: reviewer command failed" >&2
    exit 1
  }
else
  REVIEWER_OUTPUT=$(eval "$FINAL_CMD" 2>&1) || {
    echo "Error: reviewer command failed" >&2
    exit 1
  }
fi

# Determine verdict: check fail pattern first, then pass pattern
VERDICT="fail"
if echo "$REVIEWER_OUTPUT" | grep -qiE "$FAIL_PATTERN"; then
  VERDICT="fail"
elif echo "$REVIEWER_OUTPUT" | grep -qiE "$PASS_PATTERN"; then
  VERDICT="pass"
else
  # Neither matched — default to fail (safe)
  VERDICT="fail"
fi

# Check escalation
ESCALATE="false"
if [ "$VERDICT" = "fail" ] && [ "$ROUND" -ge "$MAX_ROUNDS" ]; then
  ESCALATE="true"
fi

# Escape findings for JSON
FINDINGS_ESCAPED=$(echo "$REVIEWER_OUTPUT" | sed 's/\\/\\\\/g' | sed 's/"/\\"/g' | sed ':a;N;$!ba;s/\n/\\n/g')

# Output JSON
cat <<ENDJSON
{"verdict":"${VERDICT}","round":${ROUND},"max_rounds":${MAX_ROUNDS},"escalate":${ESCALATE},"findings":"${FINDINGS_ESCAPED}"}
ENDJSON
```

---
name: autoloop
description: Automated external review loop — invoke external CLI reviewer, fix issues, resubmit until pass
type: tool
phase: implement
input: Completed work (code changes, design docs, etc.)
output: Review verdict (pass/fail) with findings
next: continue current workflow
triggers:
  - after implementation tasks complete
  - after brainstorm design finishes
  - when user requests external review
---

# Autoloop — Automated External Review

Invoke an external CLI reviewer (Codex, OpenCode, Aider, etc.) in an automated fix-resubmit loop. The script handles mechanics (invoke CLI, parse pass/fail, enforce round limits). You handle judgment (what to fix, how to fix it).

## Setup

Reviewer configs live in `reviewers/` next to this file. Each is a JSON file:

```json
{
  "name": "codex",
  "command": "codex -q --prompt '{prompt}'",
  "scope": "diff",
  "max_rounds": 3,
  "pass_pattern": "LGTM|no issues|approved|pass",
  "fail_pattern": "needs changes|issues found|fail|reject"
}
```

Copy `codex.example.json` and adjust for your reviewer. The `command` field supports placeholders:
- `{prompt}` — replaced with review prompt + code context
- `{files}` — replaced with changed file paths
- `{stdin}` — pipe code context via stdin instead of in the prompt

## Protocol

### Step 1: Prompt the user

When your work is ready for review, ask:

> "Work is ready for external review. Which reviewer would you like to use?"

List available reviewers by checking `reviewers/*.json` (exclude `.example.json` files). If none exist, tell the user to copy an example config and customize it.

### Step 2: Ask scope

> "What should I send for review?"

Options:
- **diff** (default) — git diff of current changes
- **files** — full content of changed files
- **custom** — you select specific files or code to send

### Step 3: Run the loop

Execute the script:

```bash
bash scripts/autoloop.sh --reviewer <name> --round 1 --scope <scope>
```

The script returns JSON:

```json
{
  "verdict": "pass|fail",
  "round": 1,
  "max_rounds": 3,
  "escalate": false,
  "findings": "reviewer output text"
}
```

### Step 4: Handle the result

- **verdict = pass** → Report success to user. Continue with workflow.
- **verdict = fail, escalate = false** → Read `findings`. Fix the issues. Re-run with `--round N+1`.
- **verdict = fail, escalate = true** → STOP. Show all findings to the user. Ask how to proceed. Do NOT continue fixing on your own.

## Hard Rules

1. **Never skip a round** — always re-run the script after fixing issues
2. **Never argue with findings** — fix what the reviewer says or escalate to the user
3. **Never modify the verdict** — the script decides pass/fail, not you
4. **Never exceed max rounds** — when escalate is true, stop and defer to the user
```
